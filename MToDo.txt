○BasePointAttributeの修正
・Type, Scale, NormalをAttributeで設定しておく

volumePosList[0] = v@P;
volumeScaleList[0] = v@scale;
volumeMatrixList[0] = v@RotateMatrix;
volumeFrontDirDataList[0] = v@FrontDirData;
volumeTypeList[0] = i@VolumeType;

setpointattrib(0, "scale", ptnum, scale);
setpointattrib(0, "FrontDirData", ptnum, frontDirData);
setpointattrib(0, "VolumeType", ptnum, volumeType);
setpointattrib(0, "RotateMatrix", ptnum, mRotate);
setpointattrib(0, "N", ptnum, normal);

○AttributeTypeに応じたBasePointのみに対応するように修正する
・BlockParamIndex関連の処理を消す
・VolumeTypeを統一する
 -nextVolumeTypeをvectorにしてRangeにする
 
vector nextVolumeType = point(1, "NextVolumeType", paramIndex);
volumeTypeList[0] = i@VolumeType;

if(0 < nextVolumeType.x) 
{         
	push(volumePosList, blockPosOrigin);
	push(volumeScaleList, blockScaleOrigin);
	push(volumeMatrixList, blockMatrixOrigin);
	push(volumeFrontDirDataList, blockFrontDirDataOrigin);
	push(volumeBlockParamIndexList, blockParamIndexOrigin);
	int nextType =  random(dividingBlockPosOrigin * (i + loopNum)) * (nextVolumeType.y - nextVolumeType.x) + nextVolumeType.x;
	push(volumeTypeList, nextType);
}
