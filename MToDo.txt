▲Detail
○側面の下部へのDetailBlockのアセット配置ロジックの実装
・高さを統一してアセットをランダムに配置する
float width = 0;
float blockWidthBase = 10;
vector normal = @N;

int blockNum = int(width / blockWidthBase);
float blockWidth = width / blockNum;

vector spawnVector = (1, 0, 0);
if(0.5 < abs(normal.z)) spawnVector = (0, 0, 1);

vector spawnPos = @P;
spawnPos -= spawnVector * (width / 2 - blockWidth / 2)
for(int i = 0; i < blockNum; ++i)
{
	int ptNum = addpoint();
	setpointattrib(0, "N", normal);
	spawnPos += spawnVector * blockWidth;
}
