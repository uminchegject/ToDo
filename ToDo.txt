float minLength = 1;

float width = @scale.x;
float height = @scale.y;

if(minLength < width && minLength < height)
{
    int blockNum = int(height / width);
    float blockHeight = height / blockNum;
    
    vector spawnPos = @P;
    vector spawnPtDirection = (0, 0, 0);
    vector blockScale = (1.0, 1.0, 1.0);
    blockScale.y = height;
    blockScale.x = blockHeight;

    if(width < height)
    {
        spawnPos.y = @P.y - height / 2 + blockHeight / 2;
        spawnPos.x = @P.x;
        spawnPtDirection.y = blockHeight;
        spawnPtDirection.x = 0;
        blockScale.x = width;
        blockScale.y = blockHeight;
        blockScale.z = 1;
    }
    
    for(int i = 0; i < blockNum; ++i)
    {
        int ptNum = addpoint(0, spawnPos);
        setpointattrib(0, "N", ptNum, @N);
        setpointattrib(0, "scale", ptNum, blockScale);
        spawnPos += spawnPtDirection;
    }
        
    @B = blockNum;
}
